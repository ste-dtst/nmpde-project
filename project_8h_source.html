22// This project is based on Step 26 of the deal.II tutorial.
23// Instead of using Rothe's method, we will use the method of lines
24// to solve the heat equation. The boundary conditions will be imposed
25// via Nitsche's method. For the integration in time, we will rely
26// on the ARKode package that is part of the SUNDIALS suite.
27// Adaptive mesh refinement will be implemented via solution transfer.

78 // We store the parameters for the problem in a new struct,
79 // using a ParameterHandler object to read them from an input file.
80 template <int dim>
81 struct HeatParameters
82 {
83 HeatParameters()
84 {
85 // These are the parameters. Check below for more info
86 // on their meaning.
87 prm.enter_subsection("Heat equation functions");
88 {
89 prm.add_parameter(
90 "Exact solution expression",
91 exact_solution_expression,
92 "If you don't know the solution, write unknown below.");
93 prm.add_parameter("Right hand side expression", rhs_expression);
94 prm.add_parameter("Initial value expression", initial_expression);
95 prm.add_parameter("Boundary value expression", boundary_expression);
96 }
97 prm.leave_subsection();
98
99 prm.enter_subsection("FEM parameters");
100 {
101 prm.add_parameter("Finite element degree", fe_degree);
102 prm.add_parameter("Initial global refinement", initial_refinement);
103 prm.add_parameter("Refinement threshold", refinement_threshold);
104 prm.add_parameter("Refinement top fraction", refinement_top_fraction);
105 prm.add_parameter("Refinement bottom fraction",
106 refinement_bottom_fraction);
107 prm.add_parameter("Minimum refinement level", min_refinement);
108 prm.add_parameter("Maximum refinement level", max_refinement);
109 prm.add_parameter("Gamma for Nitsche's method", gamma);
110 }
111 prm.leave_subsection();
112
113 prm.enter_subsection("ARKode solver parameters");
114 {
115 prm.add_parameter("Initial time", initial_time);
116 prm.add_parameter("Final time", final_time);
117 prm.add_parameter("Initial step size", initial_step_size);
118 prm.add_parameter("Number of solution outputs", out_steps);
119 prm.add_parameter("Error estimations per output period", est_steps);
120 prm.add_parameter("Minimum step size", minimum_step_size);
121 prm.add_parameter("Maximum order", maximum_order);
122 prm.add_parameter("Absolute tolerance", absolute_tolerance);
123 prm.add_parameter("Relative tolerance", relative_tolerance);
124 }
125 prm.leave_subsection();
126
127 prm.enter_subsection("AMG preconditioners parameters");
128 {
129 prm.add_parameter("Aggregation threshold", aggregation_threshold);
130 }
131 prm.leave_subsection();
132
133 // prm.enter_subsection("Convergence table");
134 // convergence_table.add_parameters(prm);
135 // prm.leave_subsection();
136
137 // We try to parse the input file
138 try
139 {
140 prm.parse_input("parameters/heat_" + std::to_string(dim) + "d.prm");
141 }
142 // If .prm file does not exist, it is created
143 // with default values and then parsed
144 catch (dealii::PathSearch::ExcFileNotFound &exc)
145 {
146 std::cout << "Parameters file for " << dim << "D case not found."
147 << std::endl
148 << "I've created one for you with default values."
149 << std::endl
150 << std::endl;
151 prm.print_parameters("parameters/heat_" + std::to_string(dim) +
152 "d.prm",
153 ParameterHandler::KeepDeclarationOrder);
154 prm.parse_input("parameters/heat_" + std::to_string(dim) + "d.prm");
155 }
156 // If something else goes wrong with parsing, then the program is
157 // terminated
158
159 // Initialization of the FunctionParser objects.
160 // Remark: when a function is time-dependent,
161 // we include "t" in the list of variables at the last spot and
162 // we set to true the fourth argument "time_dependent".
163
164 // Constants
165 std::map<std::string, double> constants;
166 constants["pi"] = numbers::PI;
167
168 // We initialize exact_solution only if it is known.
169 if (exact_solution_expression != "unknown")
170 {
171 exact_solution.initialize(
172 FunctionParser<dim>::default_variable_names() + ",t",
173 {exact_solution_expression},
174 constants,
175 true);
176 sol_is_known = true;
177 }
178 else
179 sol_is_known = false;
180
181 right_hand_side.initialize(FunctionParser<dim>::default_variable_names() +
182 ",t",
183 {rhs_expression},
184 constants,
185 true);
186 initial_value.initialize(FunctionParser<dim>::default_variable_names(),
187 {initial_expression},
188 constants);
189 boundary_value.initialize(FunctionParser<dim>::default_variable_names() +
190 ",t",
191 {boundary_expression},
192 constants,
193 true);
194 }
195
196 // Default parameters:
197
198 // Expressions for each function involved in the problem
199 std::string exact_solution_expression = "0";
200 std::string rhs_expression = "0";
201 std::string initial_expression = "0";
202 std::string boundary_expression = "0";
203
204 // FE degree
205 unsigned int fe_degree = 1;
206
207 // Global refinement steps to be performed before integration
208 unsigned int initial_refinement = 3;
209
210 // Threshold for the l2 norm of the error estimator.
211 // Getting beyond this triggers adaptive mesh refinement
212 float refinement_threshold = 1.0;
213
214 // Parameters for the adaptive mesh refinement
215 double refinement_top_fraction = .3;
216 double refinement_bottom_fraction = .1;
217 double min_refinement = 1;
218 double max_refinement = 6;
219
220 // This is the gamma parameter for the penalty term in Nitsche's method
221 double gamma = 10.0;
222
223 // Parameters for the ARKode solver
224 double initial_time = 0.0;
225 double final_time = 1.0;
226 double initial_step_size = 1e-2;
227 double minimum_step_size = 1e-6;
228 unsigned int maximum_order = 5;
229 double absolute_tolerance = 1e-6;
230 double relative_tolerance = 1e-5;
231
232 // In this variable we set the number of time steps, other than
233 // initial_time, at which we want to produce a solution output
234 unsigned int out_steps = 100;
235
236 // This variable specifies how many times per "output period"
237 // we will be able to do a mesh refinement.
238 unsigned int est_steps = 1;
239
240 // Function objects for the functions involved in the problem.
241 // Since we will need to evaluate them at different times,
242 // we declare the time-dependent ones mutable, in order to
243 // use a const HeatParameters object in the main program.
244 mutable FunctionParser<dim> exact_solution;
245 FunctionParser<dim> initial_value;
246 mutable FunctionParser<dim> boundary_value;
247 mutable FunctionParser<dim> right_hand_side;
248
249 // Additional flag to check if the exact solution is known.
250 // If it is, we will compute the L2 error at each output time.
251 bool sol_is_known;
252
253 // Parameters for the AMG preconditioners
254 double aggregation_threshold = 0.02;
255
256 // mutable ParsedConvergenceTable convergence_table;
257
258 ParameterHandler prm;
259 };
