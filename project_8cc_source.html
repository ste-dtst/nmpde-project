116template <int dim>
117void
118HeatEquation<dim>::assemble_ode_matrices()
119{
120 // We assemble the mass matrix and the matrix for the implicit part
121 // of the ODE in the usual way.
122
123 // Declare the quadrature formulas
124 FEValues<dim> fe_values(fe,
125 QGauss<dim>(fe.degree + 1),
126 update_values | update_gradients |
127 update_quadrature_points | update_JxW_values);
128
129 FEFaceValues<dim> fe_face_values(fe,
130 QGauss<dim - 1>(fe.degree + 1),
131 update_values | update_gradients |
132 update_quadrature_points |
133 update_normal_vectors | update_JxW_values);
134
135 // Set the usual cell-related objects
136 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
137
138 FullMatrix<double> cell_mass_matrix(dofs_per_cell, dofs_per_cell);
139 FullMatrix<double> cell_jacobian_matrix(dofs_per_cell, dofs_per_cell);
140
141 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
142
143 // Loop over cells
144 for (const auto &cell : dof_handler.active_cell_iterators())
145 {
146 // Initialize the fe_values to current cell
147 fe_values.reinit(cell);
148
149 // Initialize the cell matrices to zero
150 cell_mass_matrix = 0;
151 cell_jacobian_matrix = 0;
152
153 // The usual loop over quadrature points
154 for (const unsigned int q_index : fe_values.quadrature_point_indices())
155 for (const unsigned int i : fe_values.dof_indices())
156 for (const unsigned int j : fe_values.dof_indices())
157 {
158 cell_mass_matrix(i, j) +=
159 (fe_values.shape_value(i, q_index) * // phi_i(x_q)
160 fe_values.shape_value(j, q_index) * // phi_j(x_q)
161 fe_values.JxW(q_index)); // dx
162
163 cell_jacobian_matrix(i, j) -=
164 (fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
165 fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
166 fe_values.JxW(q_index)); // dx
167 }
168
169 // Loop over faces at the boundary of the current cell (if there are
170 // any)
171 for (auto &face : cell->face_iterators())
172 if (face->at_boundary())
173 {
174 // Initialize the fe_face_values to current face
175 fe_face_values.reinit(cell, face);
176
177 // Compute the diameter of the face, being aware that
178 // for dim = 1 it is not defined (-> we set it to 1).
179 // Here we use a constexpr if statement (C++17).
180 double face_diam;
181 if constexpr (dim == 1)
182 face_diam = 1;
183 else
184 face_diam = face->diameter();
185
186 // Loop over face quadrature points. The formulas are a bit
187 // condensed, in order to do less arithmetic operations.
188 for (const unsigned int q_index :
189 fe_face_values.quadrature_point_indices())
190 for (const unsigned int i : fe_face_values.dof_indices())
191 for (const unsigned int j : fe_face_values.dof_indices())
192 cell_jacobian_matrix(i, j) +=
193 (((fe_face_values.shape_value(i, q_index) * // phi_i(x_q)
194 fe_face_values.shape_grad(j,
195 q_index) + // grad phi_j(x_q)
196 fe_face_values.shape_grad(i,
197 q_index) * // grad phi_i(x_q)
198 fe_face_values.shape_value(j,
199 q_index)) * // phi_j(x_q)
200 fe_face_values.normal_vector(q_index) - // n
201 par.gamma / face_diam * // gamma/h
202 fe_face_values.shape_value(i, q_index) * // phi_i(x_q)
203 fe_face_values.shape_value(j,
204 q_index)) * // phi_j(x_q)
205 fe_face_values.JxW(q_index)); // dx
206 }
207
208 // Initialize the local dof indices to current cell
209 cell->get_dof_indices(local_dof_indices);
210
211 // Distribute local to global
212 for (const unsigned int i : fe_values.dof_indices())
213 for (const unsigned int j : fe_values.dof_indices())
214 {
215 mass_matrix.add(local_dof_indices[i],
216 local_dof_indices[j],
217 cell_mass_matrix(i, j));
218 jacobian_matrix.add(local_dof_indices[i],
219 local_dof_indices[j],
220 cell_jacobian_matrix(i, j));
221 }
222 }
223}

227template <int dim>
228void
229HeatEquation<dim>::assemble_ode_explicit_part(const double t)
230{
231 // We assemble also the explicit part of the ODE in the usual way.
232
233 // First, we reinit the global vector to zero
234 explicit_part.reinit(dof_handler.n_dofs());
235
236 // Declare the quadrature formulas
237 FEValues<dim> fe_values(fe,
238 QGauss<dim>(fe.degree + 1),
239 update_values | update_gradients |
240 update_quadrature_points | update_JxW_values);
241
242 FEFaceValues<dim> fe_face_values(fe,
243 QGauss<dim - 1>(fe.degree + 1),
244 update_values | update_gradients |
245 update_quadrature_points |
246 update_normal_vectors | update_JxW_values);
247
248 // Set the cell-related objects
249 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
250
251 Vector<double> cell_explicit_part(dofs_per_cell);
252
253 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
254
255 // Set the right-hand side and boundary value's time to t
256 par.right_hand_side.set_time(t);
257 par.boundary_value.set_time(t);
258
259 // Loop over cells
260 for (const auto &cell : dof_handler.active_cell_iterators())
261 {
262 // Initialize the fe_values to current cell
263 fe_values.reinit(cell);
264
265 // Initialize the cell vector to zero
266 cell_explicit_part = 0;
267
268 // The usual loop over quadrature points
269 for (const unsigned int q_index : fe_values.quadrature_point_indices())
270 {
271 const auto &x_q = fe_values.quadrature_point(q_index);
272 for (const unsigned int i : fe_values.dof_indices())
273 cell_explicit_part(i) +=
274 (fe_values.shape_value(i, q_index) * // phi_i(x_q)
275 par.right_hand_side.value(x_q) * // f(x_q)
276 fe_values.JxW(q_index)); // dx
277 }
278
279 // Loop over faces at the boundary of the current cell (if there are
280 // any)
281 for (auto &face : cell->face_iterators())
282 if (face->at_boundary())
283 {
284 // Initialize the fe_face_values to current face
285 fe_face_values.reinit(cell, face);
286
287 // Compute the diameter of the face, being aware that
288 // for dim = 1 it is not defined (-> we set it to 1).
289 // Here we use a constexpr if statement (C++17).
290 double face_diam;
291 if constexpr (dim == 1)
292 face_diam = 1;
293 else
294 face_diam = face->diameter();
295
296 // Loop over face quadrature points. The formula is a bit
297 // condensed, in order to do less arithmetic operations.
298 for (const unsigned int q_index :
299 fe_face_values.quadrature_point_indices())
300 {
301 const auto &x_q = fe_face_values.quadrature_point(q_index);
302 for (const unsigned int i : fe_face_values.dof_indices())
303 cell_explicit_part(i) +=
304 ((par.gamma / face_diam * // gamma/h
305 fe_face_values.shape_value(i, q_index) - // phi_i(x_q)
306 fe_face_values.shape_grad(i,
307 q_index) * // grad phi_i(x_q)
308 fe_face_values.normal_vector(q_index)) * // n
309 par.boundary_value.value(x_q) * // g(x_q)
310 fe_face_values.JxW(q_index)); // dx
311 }
312 }
313
314 // Initialize the local dof indices to current cell
315 cell->get_dof_indices(local_dof_indices);
316
317 // Distribute local to global
318 for (const unsigned int i : fe_values.dof_indices())
319 explicit_part(local_dof_indices[i]) += cell_explicit_part(i);
320 }
321}

325template <int dim>
326void
327HeatEquation<dim>::refine_mesh(Vector<double> &sol,
328 const unsigned int min_grid_level,
329 const unsigned int max_grid_level)
330{
331 // If we want to refine the mesh, we have to update the dimension
332 // of matrices and vectors accordingly, update mass_matrix and
333 // jacobian_matrix, transfer the solution to the new mesh.
334 // In particular, the solution vector to be transferred is the given
335 // sol instead of solution, because the refine_mesh routine is
336 // called in ode.solver_should_restart (check further on).
337
338 // We start with the Kelly error estimation
339 Vector<float> estimated_error_per_cell(triangulation.n_active_cells());
340
341 // ---------check the call to this---------------
342 KellyErrorEstimator<dim>::estimate(
343 dof_handler,
344 QGauss<dim - 1>(fe.degree + 1),
345 std::map<types::boundary_id, const Function<dim> *>(),
346 sol,
347 estimated_error_per_cell);
348
349 // We then mark the cells that have to be refined or coarsed
350 GridRefinement::refine_and_coarsen_fixed_fraction(
351 triangulation,
352 estimated_error_per_cell,
353 par.refinement_top_fraction,
354 par.refinement_bottom_fraction);
355
356 // We clear the flags that would make the mesh too coarse or too fine
357 if (triangulation.n_levels() > max_grid_level)
358 for (const auto &cell :
359 triangulation.active_cell_iterators_on_level(max_grid_level))
360 cell->clear_refine_flag();
361 for (const auto &cell :
362 triangulation.active_cell_iterators_on_level(min_grid_level))
363 cell->clear_coarsen_flag();
364
365 // The following step is to transfer the solution to the new mesh
366 SolutionTransfer<dim> solution_trans(dof_handler);
367
368 Vector<double> previous_solution;
369 previous_solution = sol;
370 triangulation.prepare_coarsening_and_refinement();
371 solution_trans.prepare_for_coarsening_and_refinement(previous_solution);
372
373 // -----complete the implementation of the sol transfer-----
374
375 // triangulation.execute_coarsening_and_refinement();
376 // setup_ode_matrices();
377
378 // solution_trans.interpolate(previous_solution, sol);
379 // constraints.distribute(sol);
380
381 // Finally, we update mass_matrix and jacobian_matrix
382 assemble_ode_matrices();
383}

387template <int dim>
388void
389HeatEquation<dim>::solve_ode()
390{
391 // We set some additional data for the solver. In particular,
392 // we want to exploit the fact that the mass matrix and the
393 // Jacobian matrix are independent of time.
394 const double out_prd = (par.final_time - par.initial_time) / par.out_steps;
395 SUNDIALS::ARKode<Vector<double>>::AdditionalData data(
396 par.initial_time, // Initial time
397 par.final_time, // Final time
398 par.initial_step_size, // Initial step size
399 out_prd, // Output period
400 par.minimum_step_size, // Minimum step size
401 par.maximum_order, // Maximum order
402 10, // Maximum nonlinear iterations (irrelevant)
403 true, // implicit_function_is_linear
404 true, // implicit_function_is_time_independent
405 true, // mass_is_time_independent
406 3, // anderson_acceleration_subspace (irrelevant)
407 par.absolute_tolerance, // Absolute tolerance
408 par.relative_tolerance); // Relative tolerance
409
410 // Here we declare the ARKode object
411 SUNDIALS::ARKode<Vector<double>> ode(data);
412
413 // We have to tell the solver how to multiply the mass matrix
414 // by a vector v
415 ode.mass_times_vector =
416 [&](const double t, const Vector<double> &v, Vector<double> &Mv) {
417 (void)t;
418 mass_matrix.vmult(Mv, v);
419 };
420
421 // We set the explicit function
422 ode.explicit_function =
423 [&](const double t, const Vector<double> &y, Vector<double> &explicit_f) {
424 (void)y;
425 assemble_ode_explicit_part(t);
426 explicit_f = explicit_part;
427 };
428
429 // We set the implicit function
430 ode.implicit_function =
431 [&](const double t, const Vector<double> &y, Vector<double> &implicit_f) {
432 (void)t;
433 jacobian_matrix.vmult(implicit_f, y);
434 };
435
436 // Since the implicit function is linear, we can supply
437 // the Jacobian matrix directly to solve the implicit part
438 // of the ODE with a single Newton iteration.
439 ode.jacobian_times_vector = [&](const Vector<double> &v,
440 Vector<double> &Jv,
441 const double t,
442 const Vector<double> &y,
443 const Vector<double> &fy) {
444 (void)t;
445 (void)y;
446 (void)fy;
447 jacobian_matrix.vmult(Jv, v);
448 };
449
450 // When mass_matrix is not the identity matrix, we must
451 // provide a linear solver for the system Mx = b
452 // through ode.solve_mass.
453 // The default linear solver for SUNDIALS is a preconditioned
454 // version of GMRES. However, the mass matrix is symmetric and
455 // positive definite, therefore we can use a preconditioned
456 // conjugate gradient method. We have to specify this to the solver.
457 ode.solve_mass = [&](SUNDIALS::SundialsOperator<Vector<double>> &op,
458 SUNDIALS::SundialsPreconditioner<Vector<double>> &prec,
459 Vector<double> &x,
460 const Vector<double> &b,
461 double tol) {
462 // We forget about op and prec, since we want to provide
463 // our custom preconditioner and solver
464 (void)op;
465 (void)prec;
466
467 // We provide the solver and the preconditioner in the
468 // same way as we have done in previous tutorial programs
469 SolverControl solver_control(1000, tol);
470 SolverCG<Vector<double>> solver(solver_control);
471
472 PreconditionSSOR<SparseMatrix<double>> preconditioner;
473 preconditioner.initialize(mass_matrix, 1.2);
474
475 solver.solve(mass_matrix, x, b, preconditioner);
476 };
477
478 // Also the matrix of the linearized system is
479 // positive definite for any legal choice of gamma.
480 // Therefore, we should use CG also for the implicit part of the ODE.
481 // However, I've encountered some problems in supplying the
482 // ode.solve_linearized_system function, see temp_dummy.cc
483 // in the directory other_files.
484
485 // Here we store some relevant info on the time stepping
486 long int nsteps = 0; // Number of steps taken in the solver
487 double hlast =
488 par.initial_step_size; // Step size taken on the last internal step
489 // double hcur = hlast; // Step size to be attempted on the next
490 // internal step
491
492 // The following function is deputed to decide whether
493 // a mesh refinement is needed at time t. In that case,
494 // it will call refine_mesh() and return true in order to
495 // trigger the reset() function of the ARKode solver.
496 // Otherwise, it will return false.
497 // ode.solver_should_restart = [&](const double t, Vector<double> &sol){
498 // (void)t;
499 // (void)sol;
500 // // Find a satisfying condition, e.g. every 10 steps?
501 // if (false)
502 // {
503 // refine_mesh(sol, par.min_refinement, par.max_refinement);
504 // return true;
505 // }
506 // else
507 // return false;
508 // };
509
510 // We supply a function for the output at fixed time intervals
511 ode.output_step = [&](const double t,
512 const Vector<double> &sol,
513 const unsigned int step_number) {
514 std::cout << "-------------------------------------------" << std::endl;
515 std::cout << "Time step " << step_number << " at t = " << t << "."
516 << std::endl;
517 std::cout << "L_inf norm of solution: " << solution.linfty_norm()
518 << std::endl;
519
520 // Some info on the time stepping so far. Here we call
521 // some ARKode functions that are not wrapped by deal.II
522 ARKStepGetNumSteps(ode.get_arkode_memory(), &nsteps);
523 ARKStepGetLastStep(ode.get_arkode_memory(), &hlast);
524 std::cout << "Number of ARKode steps taken so far: " << nsteps << std::endl;
525 std::cout << "Step size taken on the last internal step: " << hlast
526 << std::endl;
527
528 // // Some info on the mesh and on the number of degrees of freedom
529 // std::cout << "Number of active cells: " << triangulation.n_active_cells()
530 // << std::endl
531 // << "Number of degrees of freedom: " << dof_handler.n_dofs()
532 // << std::endl;
533
534 // If the exact solution is known, we compute the L2 error at time t
535 if (par.sol_is_known)
536 {
537 par.exact_solution.set_time(t);
538 Vector<double> error(solution.size());
539 VectorTools::integrate_difference(dof_handler,
540 solution,
541 par.exact_solution,
542 error,
543 QGauss<dim>(fe.degree + 1),
544 VectorTools::L2_norm);
545 std::cout << "L2 error: " << error.l2_norm() << std::endl;
546 }
547
548 // We output the solution at the current time step
549 output_results(sol, step_number);
550 };
551
552 // Finally, we are ready to solve the ODE
553 ode.solve_ode(solution);
554}

560HeatEquation<dim>::output_results(const Vector<double> &sol,
561 const unsigned int step_no) const
562{
563 DataOut<dim> data_out;

567

575
576 static std::vector<std::pair<double, std::string>> times_and_names;
577 times_and_names.push_back({step_no, filename});
578
579 std::ofstream pvd_output(dirname + "solution.pvd");
580
581 DataOutBase::write_pvd_record(pvd_output, times_and_names);
582}

586template <int dim>
587void
588HeatEquation<dim>::run()
589{
590 // Mesh generation, DoF distribution and sparsity pattern allocation
591 if (dim == 1)
592 GridGenerator::hyper_cube(triangulation, -1, 1);
593 else
594 GridGenerator::hyper_L(triangulation);
595 triangulation.refine_global(par.initial_refinement);
596
597 setup_ode();
598
599 // Assembly of the mass matrix and the implicit part of the ODE
600 assemble_ode_matrices();
601
602 // We set the initial condition
603 VectorTools::interpolate(dof_handler, par.initial_value, solution);
604
605 // ODE solution
606 solve_ode();
