138 template <int dim>
139 double RightHandSide<dim>::value(const Point<dim> & p,
140 const unsigned int component) const
141 {
142 (void)component;
143 AssertIndexRange(component, 1);
144 Assert(dim == 2, ExcNotImplemented());
145
146 const double time = this->get_time();
147 const double point_within_period =
148 (time / period - std::floor(time / period));
149
150 if ((point_within_period >= 0.0) && (point_within_period <= 0.2))
151 {
152 if ((p[0] > 0.5) && (p[1] > -0.5))
153 return 1;
154 else
155 return 0;
156 }
157 else if ((point_within_period >= 0.5) && (point_within_period <= 0.7))
158 {
159 if ((p[0] > -0.5) && (p[1] > 0.5))
160 return 1;
161 else
162 return 0;
163 }
164 else
165 return 0;
166 }

237 template <int dim>
238 void HeatEquation<dim>::assemble_ode_matrices()
239 {
240 // We assemble the mass matrix and the matrix for the implicit part
241 // of the ODE in the usual way.
242
243 // Declare the quadrature formulas
244 FEValues<dim> fe_values(fe,
245 QGauss<dim>(fe.degree + 1),
246 update_values | update_gradients |
247 update_quadrature_points | update_JxW_values);
248
249 FEFaceValues<dim> fe_face_values(fe,
250 QGauss<dim-1>(fe.degree + 1),
251 update_values | update_gradients |
252 update_quadrature_points | update_normal_vectors |
253 update_JxW_values);
254
255 // Set the usual cell-related objects
256 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
257
258 FullMatrix<double> cell_mass_matrix(dofs_per_cell, dofs_per_cell);
259 FullMatrix<double> cell_jacobian_matrix(dofs_per_cell, dofs_per_cell);
260
261 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
262
263 // Loop over cells
264 for (const auto &cell : dof_handler.active_cell_iterators())
265 {
266 // Initialize the fe_values to current cell
267 fe_values.reinit(cell);
268
269 // Initialize the cell matrices to zero
270 cell_mass_matrix = 0;
271 cell_jacobian_matrix = 0;
272
273 // The usual loop over quadrature points
274 for (const unsigned int q_index : fe_values.quadrature_point_indices())
275 for (const unsigned int i : fe_values.dof_indices())
276 for (const unsigned int j : fe_values.dof_indices())
277 {
278 cell_mass_matrix(i, j) += (fe_values.shape_value(i, q_index) * // phi_i(x_q)
279 fe_values.shape_value(j, q_index) * // phi_j(x_q)
280 fe_values.JxW(q_index)); // dx
281
282 cell_jacobian_matrix(i, j) -= (fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
283 fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
284 fe_values.JxW(q_index)); // dx
285 }
286
287 // Loop over faces at the boundary of the current cell (if there are any)
288 for (auto &face: cell->face_iterators())
289 if (face->at_boundary())
290 {
291 // Initialize the fe_face_values to current face
292 fe_face_values.reinit(cell, face);
293
294 // Loop over face quadrature points. The formulas are a bit condensed,
295 // in order to do less arithmetic operations.
296 for (const unsigned int q_index : fe_face_values.quadrature_point_indices())
297 for (const unsigned int i : fe_face_values.dof_indices())
298 for (const unsigned int j : fe_face_values.dof_indices())
299 cell_jacobian_matrix(i, j) += (((fe_face_values.shape_value(i, q_index) * // phi_i(x_q)
300 fe_face_values.shape_grad(j, q_index) + // grad phi_j(x_q)
301 fe_face_values.shape_grad(i, q_index) * // grad phi_i(x_q)
302 fe_face_values.shape_value(j, q_index)) * // phi_j(x_q)
303 fe_face_values.normal_vector(q_index) - // n
304 gamma / (face->diameter()) * // gamma/h
305 fe_face_values.shape_value(i, q_index) * // phi_i(x_q)
306 fe_face_values.shape_value(j, q_index)) * // phi_j(x_q)
307 fe_face_values.JxW(q_index)); // dx
308 }
309
310 // Initialize the local dof indices to current cell
311 cell->get_dof_indices(local_dof_indices);
312
313 // Distribute local to global
314 for (const unsigned int i : fe_values.dof_indices())
315 for (const unsigned int j : fe_values.dof_indices())
316 {
317 mass_matrix.add(local_dof_indices[i],
318 local_dof_indices[j],
319 cell_mass_matrix(i, j));
320 jacobian_matrix.add(local_dof_indices[i],
321 local_dof_indices[j],
322 cell_jacobian_matrix(i, j));
323 }
324 }
325 }

329 template <int dim>
330 void HeatEquation<dim>::assemble_ode_explicit_part(const double t)
331 {
332 // We assemble also the explicit part of the ODE in the usual way.
333
334 // First, we reinit the global vector to zero
335 explicit_part.reinit(dof_handler.n_dofs());
336
337 // Declare the quadrature formulas
338 FEValues<dim> fe_values(fe,
339 QGauss<dim>(fe.degree + 1),
340 update_values | update_gradients |
341 update_quadrature_points | update_JxW_values);
342
343 FEFaceValues<dim> fe_face_values(fe,
344 QGauss<dim-1>(fe.degree + 1),
345 update_values | update_gradients |
346 update_quadrature_points | update_normal_vectors |
347 update_JxW_values);
348
349 // Set the cell-related objects
350 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
351
352 Vector<double> cell_explicit_part(dofs_per_cell);
353
354 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
355
356 // Declare the right-hand side and boundary values objects
357 RightHandSide<dim> right_hand_side;
358 BoundaryValues<dim> boundary_values;
359
360 // Set these objects' time to t
361 right_hand_side.set_time(t);
362 boundary_values.set_time(t);
363
364 // Loop over cells
365 for (const auto &cell : dof_handler.active_cell_iterators())
366 {
367 // Initialize the fe_values to current cell
368 fe_values.reinit(cell);
369
370 // Initialize the cell vector to zero
371 cell_explicit_part = 0;
372
373 // The usual loop over quadrature points
374 for (const unsigned int q_index : fe_values.quadrature_point_indices())
375 {
376 const auto &x_q = fe_values.quadrature_point(q_index);
377 for (const unsigned int i : fe_values.dof_indices())
378 cell_explicit_part(i) += (fe_values.shape_value(i, q_index) * // phi_i(x_q)
379 right_hand_side.value(x_q) * // f(x_q)
380 fe_values.JxW(q_index)); // dx
381 }
382
383 // Loop over faces at the boundary of the current cell (if there are any)
384 for (auto &face: cell->face_iterators())
385 if (face->at_boundary())
386 {
387 // Initialize the fe_face_values to current face
388 fe_face_values.reinit(cell, face);
389
390 // Loop over face quadrature points. The formula is a bit condensed,
391 // in order to do less arithmetic operations.
392 for (const unsigned int q_index : fe_face_values.quadrature_point_indices())
393 {
394 const auto &x_q = fe_face_values.quadrature_point(q_index);
395 for (const unsigned int i : fe_face_values.dof_indices())
396 cell_explicit_part(i) += ((gamma / (face->diameter()) * // gamma/h
397 fe_face_values.shape_value(i, q_index) - // phi_i(x_q)
398 fe_face_values.shape_grad(i, q_index) * // grad phi_i(x_q)
399 fe_face_values.normal_vector(q_index)) * // n
400 boundary_values.value(x_q) * // g(x_q)
401 fe_face_values.JxW(q_index)); // dx
402 }
403 }
404
405 // Initialize the local dof indices to current cell
406 cell->get_dof_indices(local_dof_indices);
407
408 // Distribute local to global
409 for (const unsigned int i : fe_values.dof_indices())
410 explicit_part(local_dof_indices[i]) += cell_explicit_part(i);
411 }
412 }

416 template <int dim>
417 void HeatEquation<dim>::solve_ode()
418 {
419 // We set some additional data for the solver. In particular,
420 // we want to exploit the fact that the mass matrix and the
421 // Jacobian matrix are independent of time.
422 const double out_prd = (final_time - initial_time)/nsteps;
423 SUNDIALS::ARKode<Vector<double>>::AdditionalData data(initial_time,
424 final_time,
425 1e-2, // Initial step size
426 out_prd, // Output period
427 min_step, // Minimum step size
428 5, // Maximum order
429 10, // Maximum nonlinear iterations
430 true, // implicit_function_is_linear
431 true, // implicit_function_is_time_independent
432 true); // mass_is_time_independent
433
434 // Here we declare the ARKode object
435 SUNDIALS::ARKode<Vector<double>> ode(data);
436
437 // We have to tell the solver how to multiply the mass matrix
438 // by a vector v
439 ode.mass_times_vector = [&] (const double t,
440 const Vector<double> &v,
441 Vector<double> &Mv)
442 {
443 (void)t;
444 mass_matrix.vmult(Mv, v);
445 };
446
447 // We set the explicit function
448 ode.explicit_function = [&] (const double t,
449 const Vector<double> &y,
450 Vector<double> &explicit_f)
451 {
452 (void)y;
453 assemble_ode_explicit_part(t);
454 explicit_f = explicit_part;
455 };
456
457 // We set the implicit function
458 ode.implicit_function = [&] (const double t,
459 const Vector<double> &y,
460 Vector<double> &implicit_f)
461 {
462 (void)t;
463 jacobian_matrix.vmult(implicit_f,y);
464 };
465
466 // Since the implicit function is linear, we can supply
467 // the Jacobian matrix directly to solve the implicit part
468 // of the ODE with a single Newton iteration.
469 ode.jacobian_times_vector = [&] (const Vector<double> &v,
470 Vector<double> &Jv,
471 const double t,
472 const Vector<double> &y,
473 const Vector<double> &fy)
474 {
475 (void)t;
476 (void)y;
477 (void)fy;
478 jacobian_matrix.vmult(Jv, v);
479 };
480
481 // When mass_matrix is not the identity matrix, we must
482 // provide a linear solver for the system Mx = b
483 // through ode.solve_mass.
484 // The default linear solver for SUNDIALS is a preconditioned
485 // version of GMRES. However, the mass matrix is symmetric and
486 // positive definite, therefore we can use a preconditioned
487 // conjugate gradient method. We have to specify this to the solver.
488 ode.solve_mass = [&](SUNDIALS::SundialsOperator<Vector<double>> &op,
489 SUNDIALS::SundialsPreconditioner<Vector<double>> &prec,
490 Vector<double> &x,
491 const Vector<double> &b,
492 double tol)
493 {
494 // We forget about op and prec, since we want to provide
495 // our custom preconditioner and solver
496 (void) op;
497 (void) prec;
498
499 // We provide the solver and the preconditioner in the
500 // same way as we have done in previous tutorial programs
501 SolverControl solver_control(1000, tol);
502 SolverCG<Vector<double>> solver(solver_control);
503
504 PreconditionSSOR<SparseMatrix<double>> preconditioner;
505 preconditioner.initialize(mass_matrix, 1.2);
506
507 solver.solve(mass_matrix, x, b, preconditioner);
508 };
509
510 // QUESTION: is also the Jacobian matrix positive definite for
511 // appropriate values of gamma? If so, we could use CG
512 // also for the implicit part of the ODE.
513 // However, I've encountered some problems in supplying the
514 // ode.solve_linearized_system function, see temp_dummy.cc
515 // in the directory other_files.
516
517 // We supply a function for the output at fixed time intervals
518 ode.output_step = [&](const double t,
519 const Vector<double> &sol,
520 const unsigned int step_number)
521 {
522 std::cout << "Time step " << step_number << " at t = "
523 << t << "." << std::endl;
524 std::cout << "l_inf norm of solution: " << solution.linfty_norm() << std::endl;
525
526 output_results(sol, step_number);
527 };
528
529 // Finally, we are ready to solve the ODE
530 ode.solve_ode(solution);
531 }

561 template <int dim>
562 void HeatEquation<dim>::run()
563 {
564 // Mesh generation, DoF distribution and sparsity pattern allocation
565 GridGenerator::hyper_L(triangulation);
566 triangulation.refine_global(initial_global_refinement);
567
568 setup_ode();
569
570 // Assembly of the mass matrix and the implicit part of the ODE
571 assemble_ode_matrices();
572
573 // For a simpler test, we set the initial condition to zero
574 VectorTools::interpolate(dof_handler,
575 Functions::ZeroFunction<dim>(),
576 solution);
577
578 // ODE solution
579 solve_ode();
580 }
