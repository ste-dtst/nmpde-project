84template <int dim>
85void
86HeatEquation<dim>::setup_ode()
87{
88 // As usual, we distribute the degrees of freedom
89 dof_handler.distribute_dofs(fe);
90
91 std::cout << std::endl
92 << "===========================================" << std::endl
93 << "Number of active cells: " << triangulation.n_active_cells()
94 << std::endl
95 << "Number of degrees of freedom: " << dof_handler.n_dofs()
96 << std::endl
97 << std::endl;
98
99 // We reinitialize the constraints object
100 constraints.clear();
101 DoFTools::make_hanging_node_constraints(dof_handler, constraints);
102 constraints.close();
103
104 // We allocate the sparsity pattern for the matrices
105 DynamicSparsityPattern dsp(dof_handler.n_dofs());
106 DoFTools::make_sparsity_pattern(dof_handler, dsp, constraints);
107 sparsity_pattern.copy_from(dsp);
108
109 // Finally, we initialize the matrices and the solution vector
110 mass_matrix.reinit(sparsity_pattern);
111 jacobian_matrix.reinit(sparsity_pattern);
112
113 solution.reinit(dof_handler.n_dofs());
114}

118template <int dim>
119void
120HeatEquation<dim>::assemble_ode_matrices()
121{
122 // We assemble the mass matrix and the matrix for the implicit part
123 // of the ODE using the MeshWorker framework.
124 // First, we need the ScratchData and CopyData objects.
125
126 // Scratch data: declare the quadrature formulas
127 MeshWorker::ScratchData<dim> scratch_data(fe,
128 QGauss<dim>(fe.degree + 1),
129 update_values | update_gradients |
130 update_quadrature_points |
131 update_JxW_values,
132 QGauss<dim - 1>(fe.degree + 1),
133 update_values | update_gradients |
134 update_quadrature_points |
135 update_normal_vectors |
136 update_JxW_values);
137
138 // Copy data: we need two matrices to store the local contributions
139 // to the mass matrix and the jacobian matrix, no vector.
140 MeshWorker::CopyData<2, 0, 1> copy_data(fe.n_dofs_per_cell());
141
142 // Cell worker
143 const auto cell_worker = [&](const auto &cell, auto &scratch, auto &copy) {
144 // Initialize the fe_values to current cell
145 const auto &fe_values = scratch.reinit(cell);
146
147 // Get the cell matrices, cell explicit part, and local dof indices
148 auto &cell_mass_matrix = copy.matrices[0];
149 auto &cell_jacobian_matrix = copy.matrices[1];
150 auto &local_dof_indices = copy.local_dof_indices[0];
151
152 // Initialize the matrices to zero
153 cell_mass_matrix = 0;
154 cell_jacobian_matrix = 0;
155
156 // Initialize the local dof indices to current cell
157 cell->get_dof_indices(local_dof_indices);
158
159 // The usual loop over quadrature points, with a slight variation.
160 // Since the matrices are symmetric, we do the calculations only
161 // for the lower triangular part and save some computational work.
162 for (const unsigned int q_index : fe_values.quadrature_point_indices())
163 for (const unsigned int i : fe_values.dof_indices())
164 for (unsigned int j = 0; j <= i; ++j)
165 {
166 // Do the calculations
167 auto mass_temp = (fe_values.shape_value(i, q_index) * // phi_i(x_q)
168 fe_values.shape_value(j, q_index) * // phi_j(x_q)
169 fe_values.JxW(q_index)); // dx
170
171 auto jacobian_temp =
172 (fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
173 fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
174 fe_values.JxW(q_index)); // dx
175
176 // Update the matrices exploiting the symmetry
177 if (j == i)
178 {
179 cell_mass_matrix(i, j) += mass_temp;
180 cell_jacobian_matrix(i, j) -= jacobian_temp;
181 }
182 else // j < i
183 {
184 cell_mass_matrix(i, j) += mass_temp;
185 cell_mass_matrix(j, i) += mass_temp;
186 cell_jacobian_matrix(i, j) -= jacobian_temp;
187 cell_jacobian_matrix(j, i) -= jacobian_temp;
188 }
189 }
190 };
191
192 // Boundary worker
193 const auto boundary_worker =
194 [&](const auto &cell, const auto &face_no, auto &scratch, auto &copy) {
195 // Initialize the fe_face_values to current face
196 auto &fe_face_values = scratch.reinit(cell, face_no);
197
198 // In the boundary worker we will work only for the jacobian matrix,
199 // therefore we create an alias only for it. We DO NOT reinitialize
200 // anything here, or we will overwrite data computed in the cell worker.
201 auto &cell_jacobian_matrix = copy.matrices[1];
202
203 // Since we already have initialized the local dof indices in the cell
204 // worker, we don't need to do it again here.
205
206 // Compute the diameter of the face, being aware that
207 // for dim = 1 it is not defined (-> we set it to 1).
208 // Here we use a constexpr if statement (C++17).
209 double face_diam;
210 if constexpr (dim == 1)
211 face_diam = 1;
212 else
213 face_diam = cell->face(face_no)->diameter();
214
215 // Loop over face quadrature points. The formulas are a bit
216 // condensed, in order to do less arithmetic operations.
217 // As before, we also exploit the symmetry of the matrices.
218 for (const unsigned int q_index :
219 fe_face_values.quadrature_point_indices())
220 for (const unsigned int i : fe_face_values.dof_indices())
221 for (unsigned int j = 0; j <= i; ++j)
222 {
223 // Do the calculations
224 auto jacobian_temp =
225 (((fe_face_values.shape_value(i, q_index) * // phi_i(x_q)
226 fe_face_values.shape_grad(j, q_index) + // grad phi_j(x_q)
227 fe_face_values.shape_grad(i, q_index) * // grad phi_i(x_q)
228 fe_face_values.shape_value(j, q_index)) * // phi_j(x_q)
229 fe_face_values.normal_vector(q_index) - // n
230 par.gamma / face_diam * // gamma/h
231 fe_face_values.shape_value(i, q_index) * // phi_i(x_q)
232 fe_face_values.shape_value(j, q_index)) * // phi_j(x_q)
233 fe_face_values.JxW(q_index)); // dx
234
235 // Update the matrix exploiting the symmetry
236 if (j == i)
237 cell_jacobian_matrix(i, j) += jacobian_temp;
238 else // j < i
239 {
240 cell_jacobian_matrix(i, j) += jacobian_temp;
241 cell_jacobian_matrix(j, i) += jacobian_temp;
242 }
243 }
244 };
245
246 // Copier function
247 const auto copier = [&](const auto &copy) {
248 // auto &cell_mass_matrix = copy.matrices[0];
249 // auto &cell_jacobian_matrix = copy.matrices[1];
250 // auto &local_dof_indices = copy.local_dof_indices[0];
251
252 // Distribute local to global
253 constraints.distribute_local_to_global(copy.matrices[0],
254 copy.local_dof_indices[0],
255 mass_matrix);
256 constraints.distribute_local_to_global(copy.matrices[1],
257 copy.local_dof_indices[0],
258 jacobian_matrix);
259 };
260
261 // Finally, we run the MeshWorker loop
262 MeshWorker::mesh_loop(dof_handler.begin_active(),
263 dof_handler.end(),
264 cell_worker,
265 copier,
266 scratch_data,
267 copy_data,
268 MeshWorker::assemble_own_cells |
269 MeshWorker::assemble_boundary_faces,
270 boundary_worker);
271}

275template <int dim>
276void
277HeatEquation<dim>::assemble_ode_explicit_part(const double t)
278{
279 // We assemble also the explicit part of the ODE with MeshWorker.
280
281 // First, we reinit the global vector to zero
282 explicit_part.reinit(dof_handler.n_dofs());
283
284 // Set the right-hand side and boundary value's time to t
285 par.right_hand_side.set_time(t);
286 par.boundary_value.set_time(t);
287
288 // Scratch data: declare the quadrature formulas
289 MeshWorker::ScratchData<dim> scratch_data(fe,
290 QGauss<dim>(fe.degree + 1),
291 update_values | update_gradients |
292 update_quadrature_points |
293 update_JxW_values,
294 QGauss<dim - 1>(fe.degree + 1),
295 update_values | update_gradients |
296 update_quadrature_points |
297 update_normal_vectors |
298 update_JxW_values);
299
300 // Copy data: we need only a vector to store the local contributions
301 // to the explicit part of the ODE.
302 MeshWorker::CopyData<0, 1, 1> copy_data(fe.n_dofs_per_cell());
303
304 // Cell worker
305 const auto cell_worker = [&](const auto &cell, auto &scratch, auto &copy) {
306 // Initialize the fe_values to current cell
307 const auto &fe_values = scratch.reinit(cell);
308
309 // Get the cell explicit part and the local dof indices
310 auto &cell_explicit_part = copy.vectors[0];
311 auto &local_dof_indices = copy.local_dof_indices[0];
312
313 // Initialize the vector to zero
314 cell_explicit_part = 0;
315
316 // Initialize the local dof indices to current cell
317 cell->get_dof_indices(local_dof_indices);
318
319 // The usual loop over quadrature points
320 for (const unsigned int q_index : fe_values.quadrature_point_indices())
321 {
322 const auto &x_q = fe_values.quadrature_point(q_index);
323 for (const unsigned int i : fe_values.dof_indices())
324 cell_explicit_part(i) +=
325 (fe_values.shape_value(i, q_index) * // phi_i(x_q)
326 par.right_hand_side.value(x_q) * // f(x_q)
327 fe_values.JxW(q_index)); // dx
328 }
329 };
330
331 // Boundary worker
332 const auto boundary_worker =
333 [&](const auto &cell, const auto &face_no, auto &scratch, auto &copy) {
334 // Initialize the fe_face_values to current face
335 auto &fe_face_values = scratch.reinit(cell, face_no);
336
337 // As done before, we DO NOT reinitialize anything here, or
338 // we will overwrite data computed in the cell worker.
339 auto &cell_explicit_part = copy.vectors[0];
340
341 // Since we already have initialized the local dof indices in the cell
342 // worker, we don't need to do it again here.
343
344 // Compute the diameter of the face, being aware that
345 // for dim = 1 it is not defined (-> we set it to 1).
346 // Here we use a constexpr if statement (C++17).
347 double face_diam;
348 if constexpr (dim == 1)
349 face_diam = 1;
350 else
351 face_diam = cell->face(face_no)->diameter();
352
353 // Loop over face quadrature points. The formula is a bit
354 // condensed, in order to do less arithmetic operations.
355 for (const unsigned int q_index :
356 fe_face_values.quadrature_point_indices())
357 {
358 const auto &x_q = fe_face_values.quadrature_point(q_index);
359 for (const unsigned int i : fe_face_values.dof_indices())
360 cell_explicit_part(i) +=
361 ((par.gamma / face_diam * // gamma/h
362 fe_face_values.shape_value(i, q_index) - // phi_i(x_q)
363 fe_face_values.shape_grad(i, q_index) * // grad phi_i(x_q)
364 fe_face_values.normal_vector(q_index)) * // n
365 par.boundary_value.value(x_q) * // g(x_q)
366 fe_face_values.JxW(q_index)); // dx
367 }
368 };
369
370 // Copier function
371 const auto copier = [&](const auto &copy) {
372 // auto &cell_explicit_part = copy.vectors[0];
373 // auto &local_dof_indices = copy.local_dof_indices[0];
374
375 // Distribute local to global
376 constraints.distribute_local_to_global(copy.vectors[0],
377 copy.local_dof_indices[0],
378 explicit_part);
379 };
380
381 // Finally, we run the MeshWorker loop
382 MeshWorker::mesh_loop(dof_handler.begin_active(),
383 dof_handler.end(),
384 cell_worker,
385 copier,
386 scratch_data,
387 copy_data,
388 MeshWorker::assemble_own_cells |
389 MeshWorker::assemble_boundary_faces,
390 boundary_worker);
391}

395template <int dim>
396void
397HeatEquation<dim>::refine_mesh(Vector<double> &sol,
398 const unsigned int min_grid_level,
399 const unsigned int max_grid_level)
400{
401 // If we want to refine the mesh, we have to update the dimension
402 // of matrices and vectors accordingly, update mass_matrix and
403 // jacobian_matrix, transfer the solution to the new mesh.
404 // In particular, the solution vector to be transferred is the given
405 // sol instead of solution, because the refine_mesh routine is
406 // called in ode.solver_should_restart (check further on).
407
408 // We mark the cells that have to be refined or coarsed
409 GridRefinement::refine_and_coarsen_fixed_fraction(
410 triangulation,
411 estimated_error_per_cell,
412 par.refinement_top_fraction,
413 par.refinement_bottom_fraction);
414
415 // We clear the flags that would make the mesh too coarse or too fine
416 if (triangulation.n_levels() > max_grid_level)
417 for (const auto &cell :
418 triangulation.active_cell_iterators_on_level(max_grid_level))
419 cell->clear_refine_flag();
420 for (const auto &cell :
421 triangulation.active_cell_iterators_on_level(min_grid_level))
422 cell->clear_coarsen_flag();
423
424 // The following step is to transfer the solution to the new mesh
425 SolutionTransfer<dim> solution_trans(dof_handler);
426
427 Vector<double> previous_solution;
428 previous_solution = sol;
429 triangulation.prepare_coarsening_and_refinement();
430 solution_trans.prepare_for_coarsening_and_refinement(previous_solution);
431
432 triangulation.execute_coarsening_and_refinement();
433 setup_ode();
434
435 solution_trans.interpolate(previous_solution, sol);
436 constraints.distribute(sol);
437
438 // Finally, we update mass_matrix and jacobian_matrix
439 assemble_ode_matrices();
440}

444template <int dim>
445void
446HeatEquation<dim>::solve_ode()
447{
448 // Here we setup the ARKode solver and solve the ODE.
449 // What we will do below depends also on the refinement
450 // strategy we want to implement. In particular, we'd like
451 // to be able to refine the mesh after each solution output
452 // and also at additional timesteps between outputs. Our solver
453 // can be provided two functions to handle output and refinement:
454 // output_step and solver_should_restart. Unfortunately,
455 // they both are called after the same amount of time.
456 // Therefore, we choose to provide only solver_should_restart
457 // and have a loop in which we will call solve_ode_incrementally
458 // and output the solution. SUNDIALS will take care of
459 // calling solver_should_restart between outputs, as desired.
460
461 // At first, we have to fix the time period between two
462 // solution outputs and the one between two error estimations.
463 const double out_prd = (par.final_time - par.initial_time) / par.out_steps;
464 const double est_prd = out_prd / par.est_steps;
465
466 // We set some additional data for the solver. In particular,
467 // we want to exploit the fact that the mass matrix and the
468 // Jacobian matrix are independent of time.
469 SUNDIALS::ARKode<Vector<double>>::AdditionalData data(
470 par.initial_time, // Initial time
471 par.final_time, // Final time
472 par.initial_step_size, // Initial step size
473 est_prd, // Period between two error estimations (!)
474 par.minimum_step_size, // Minimum step size
475 par.maximum_order, // Maximum order
476 10, // Maximum nonlinear iterations (irrelevant)
477 true, // implicit_function_is_linear
478 true, // implicit_function_is_time_independent
479 true, // mass_is_time_independent
480 3, // anderson_acceleration_subspace (irrelevant)
481 par.absolute_tolerance, // Absolute tolerance
482 par.relative_tolerance); // Relative tolerance
483
484 // Here we declare the ARKode object
485 SUNDIALS::ARKode<Vector<double>> ode(data);
486
487 // We have to tell the solver how to multiply the mass matrix
488 // by a vector v
489 ode.mass_times_vector =
490 [&](const double t, const Vector<double> &v, Vector<double> &Mv) {
491 (void)t;
492 mass_matrix.vmult(Mv, v);
493 };
494
495 // We set the explicit function
496 ode.explicit_function =
497 [&](const double t, const Vector<double> &y, Vector<double> &explicit_f) {
498 (void)y;
499 assemble_ode_explicit_part(t);
500 explicit_f = explicit_part;
501 };
502
503 // We set the implicit function
504 ode.implicit_function =
505 [&](const double t, const Vector<double> &y, Vector<double> &implicit_f) {
506 (void)t;
507 jacobian_matrix.vmult(implicit_f, y);
508 };
509
510 // Since the implicit function is linear, we can supply
511 // the Jacobian matrix directly to solve the implicit part
512 // of the ODE with a single Newton iteration.
513 ode.jacobian_times_vector = [&](const Vector<double> &v,
514 Vector<double> &Jv,
515 const double t,
516 const Vector<double> &y,
517 const Vector<double> &fy) {
518 (void)t;
519 (void)y;
520 (void)fy;
521 jacobian_matrix.vmult(Jv, v);
522 };
523
524 // When mass_matrix is not the identity matrix, we must
525 // provide a linear solver for the system Mx = b
526 // through ode.solve_mass.
527 // The default linear solver for SUNDIALS is a preconditioned
528 // version of GMRES. However, the mass matrix is symmetric and
529 // positive definite, therefore we can use a preconditioned
530 // conjugate gradient method. We have to specify this to the solver.
531 ode.solve_mass = [&](SUNDIALS::SundialsOperator<Vector<double>> &op,
532 SUNDIALS::SundialsPreconditioner<Vector<double>> &prec,
533 Vector<double> &x,
534 const Vector<double> &b,
535 double tol) {
536 // We forget about op and prec, since we want to provide
537 // our custom preconditioner and solver
538 (void)op;
539 (void)prec;
540
541 // We provide the solver and the preconditioner in the
542 // same way as we have done in previous tutorial programs
543 SolverControl solver_control(1000, tol);
544 SolverCG<Vector<double>> solver(solver_control);
545
546 PreconditionSSOR<SparseMatrix<double>> preconditioner;
547 preconditioner.initialize(mass_matrix, 1.2);
548
549 solver.solve(mass_matrix, x, b, preconditioner);
550
551 constraints.distribute(x);
552 };
553
554 // Also the matrix of the linearized system is
555 // positive definite for any legal choice of gamma.
556 // Therefore, we should use CG also for the implicit part of the ODE.
557
558 // First attempt at using a custom solver for the linearized system.
559 // I'm not satisfied, because it's a hard-coded workaround.
560 //
561 // ----to be improved, using it to test adaptive mesh refinement----
562 //
563 // Since the SundialsOperator in ode.solve_linearized_system will only know
564 // how to do matrix-vector products, we need to define a custom class
565 // that represents the matrix. This class will inherit from
566 // SparseMatrix<double> and will implement a virtual method for
567 // vmult, which is necessary for the SolverCG class, based on the
568 // SundialsOperator one.
569 class MyMatrix : public SparseMatrix<double>
570 {
571 public:
572 MyMatrix(SUNDIALS::SundialsOperator<Vector<double>> &op)
573 : op(op)
574 {}
575
576 void
577 vmult(Vector<double> &dst, const Vector<double> &src) const
578 {
579 op.vmult(dst,
580 src); // Use the matrix-vector product of the SundialsOperator
581 }
582
583 private:
584 SUNDIALS::SundialsOperator<Vector<double>> &op;
585 };
586
587 // Now we use this class in the solve_linearized_system function
588 ode.solve_linearized_system =
589 [&](SUNDIALS::SundialsOperator<Vector<double>> &op,
590 SUNDIALS::SundialsPreconditioner<Vector<double>> &prec,
591 Vector<double> &x,
592 const Vector<double> &b,
593 double tol) {
594 // We forget about prec, since we want to provide our custom
595 // preconditioner
596 (void)prec;
597
598 // Define the solver control and the CG solver
599 SolverControl solver_control(1000, tol);
600 SolverCG<Vector<double>> solver(solver_control);
601
602 // Define the system matrix and reinit its sparsity pattern,
603 // which is the same as the mass matrix and the Jacobian matrix
604 MyMatrix sys_matrix(op);
605 SparsityPattern sparsity_pattern;
606 sparsity_pattern.copy_from(mass_matrix.get_sparsity_pattern());
607 sys_matrix.reinit(sparsity_pattern);
608
609 // PROBLEM: the preconditioner cannot be initialized at this point,
610 // because we only know how sys_matrix acts for products... in
611 // particular, at the moment it is a sparse zero matrix.
612 // Is there a workaround for this, other than providing a
613 // jacobian_preconditioner_solve function?
614
615 // PreconditionSSOR<> preconditioner;
616 // preconditioner.initialize(matrix, 1.2);
617
618 // Solve the system with no preconditioner
619 solver.solve(sys_matrix, x, b, PreconditionIdentity());
620
621 constraints.distribute(x);
622 };
623
624 // With the following function, we decide whether a mesh refinement
625 // is needed at time t. In that case, we call refine_mesh() and
626 // return true in order to trigger the reset() function of the
627 // ARKode solver. Otherwise, we return false.
628 // NOTA BENE: solver_should_restart is called recursively until it
629 // returns false, so we can refine the mesh multiple times, but also
630 // get stuck in a loop if we never meet the condition for returning false.
631 ode.solver_should_restart = [&](const double t, Vector<double> &sol) {
632 // We do the Kelly error estimation
633 estimated_error_per_cell.reinit(triangulation.n_active_cells());
634
635 KellyErrorEstimator<dim>::estimate(
636 dof_handler,
637 QGauss<dim - 1>(fe.degree + 1),
638 std::map<types::boundary_id, const Function<dim> *>(),
639 sol,
640 estimated_error_per_cell);
641
642 // We check if the error is too large.
643 if (estimated_error_per_cell.l2_norm() > par.refinement_threshold)
644 {
645 if (consecutive_refinements == 5)
646 {
647 // If we have already done 5 consecutive refinements,
648 // we stop the process and return false.
649 std::cout << "Too many refinements at time " << t << std::endl;
650 std::cout << "l2 norm of the error estimator: "
651 << estimated_error_per_cell.l2_norm() << std::endl;
652 consecutive_refinements = 0; // Reset the counter
653 return false;
654 }
655 else
656 {
657 // Otherwise, we refine the mesh and increase the counter
658 refine_mesh(sol, par.min_refinement, par.max_refinement);
659 ++consecutive_refinements;
660 return true;
661 }
662 }
663 else
664 {
665 // If the error is small enough, we reset the counter
666 consecutive_refinements = 0;
667 return false;
668 }
669 };
670
671 // Here we store some relevant info on the time stepping
672 double inter_time = par.initial_time; // Intermediate time
673 double curr_time = par.initial_time; // Current time
674 long int nsteps = 0; // Number of steps taken in the solver
675 double hlast =
676 par.initial_step_size; // Step size taken on the last internal step
677
678 // Finally, we are ready to solve the ODE
679 for (unsigned int step = 1; step <= par.out_steps; ++step)
680 {
681 // Solve the ODE until inter_time
682 inter_time += out_prd;
683 ode.solve_ode_incrementally(solution, inter_time);
684
685 // Give some info to the user. We will call below
686 // some ARKode functions that are not wrapped by deal.II
687 ARKStepGetCurrentTime(ode.get_arkode_memory(), &curr_time);
688 std::cout << "-------------------------------------------" << std::endl;
689 std::cout << "Time step " << step << " at t = " << curr_time << "."
690 << std::endl;
691
692 // Some info on the time stepping so far.
693 ARKStepGetNumSteps(ode.get_arkode_memory(), &nsteps);
694 ARKStepGetLastStep(ode.get_arkode_memory(), &hlast);
695 std::cout << "Number of ARKode steps taken so far: " << nsteps
696 << std::endl;
697 std::cout << "Step size taken on the last internal step: " << hlast
698 << std::endl;
699
700 // // Some info on the mesh and on the number of degrees of freedom
701 // std::cout << "Number of active cells: " <<
702 // triangulation.n_active_cells()
703 // << std::endl
704 // << "Number of degrees of freedom: " << dof_handler.n_dofs()
705 // << std::endl;
706
707 // If the exact solution is known, we compute the L2 error
708 // at current time
709 if (par.sol_is_known)
710 {
711 par.exact_solution.set_time(curr_time);
712 Vector<double> error(solution.size());
713 VectorTools::integrate_difference(dof_handler,
714 solution,
715 par.exact_solution,
716 error,
717 QGauss<dim>(fe.degree + 1),
718 VectorTools::L2_norm);
719 std::cout << "L2 error: " << error.l2_norm() << std::endl;
720 }
721
722 // We output the solution at the current time step
723 output_results(solution, step);
724 }
725}
