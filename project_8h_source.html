22// This project is based on Step 26 of the deal.II tutorial.
23// Instead of using Rothe's method, we will use the method of lines
24// to solve the heat equation. The boundary conditions will be imposed
25// via Nitsche's method. For the integration in time, we will rely
26// on the ARKode package that is part of the SUNDIALS suite.
27// Adaptive mesh refinement will be implemented via solution transfer.

68 // We store the parameters for the problem in a new struct,
69 // using a ParameterHandler object to read them from an input file.
70 template <int dim>
71 struct HeatParameters
72 {
73 HeatParameters()
74 {
75 // These are the parameters. Check below for more info
76 // on their meaning.
77 prm.enter_subsection("Heat equation functions");
78 {
79 prm.add_parameter(
80 "Exact solution expression",
81 exact_solution_expression,
82 "If you don't know the solution, write unknown below.");
83 prm.add_parameter("Right hand side expression", rhs_expression);
84 prm.add_parameter("Initial value expression", initial_expression);
85 prm.add_parameter("Boundary value expression", boundary_expression);
86 }
87 prm.leave_subsection();
88
89 prm.enter_subsection("FEM parameters");
90 {
91 prm.add_parameter("Finite element degree", fe_degree);
92 prm.add_parameter("Initial global refinement", initial_refinement);
93 prm.add_parameter("Refinement top fraction", refinement_top_fraction);
94 prm.add_parameter("Refinement bottom fraction",
95 refinement_bottom_fraction);
96 prm.add_parameter("Minimum refinement level", min_refinement);
97 prm.add_parameter("Maximum refinement level", max_refinement);
98 prm.add_parameter("Gamma for Nitsche's method", gamma);
99 }
100 prm.leave_subsection();
101
102 prm.enter_subsection("ARKode solver parameters");
103 {
104 prm.add_parameter("Initial time", initial_time);
105 prm.add_parameter("Final time", final_time);
106 prm.add_parameter("Initial step size", initial_step_size);
107 prm.add_parameter("Number of solution outputs", out_steps);
108 prm.add_parameter("Minimum step size", minimum_step_size);
109 prm.add_parameter("Maximum order", maximum_order);
110 prm.add_parameter("Absolute tolerance", absolute_tolerance);
111 prm.add_parameter("Relative tolerance", relative_tolerance);
112 }
113 prm.leave_subsection();
114
115 // prm.enter_subsection("Convergence table");
116 // convergence_table.add_parameters(prm);
117 // prm.leave_subsection();
118
119 // We try to parse the input file
120 try
121 {
122 prm.parse_input("parameters/heat_" + std::to_string(dim) + "d.prm");
123 }
124 // If .prm file does not exist, it is created
125 // with default values and then parsed
126 catch (dealii::PathSearch::ExcFileNotFound &exc)
127 {
128 std::cout << "Parameters file for " << dim << "D case not found."
129 << std::endl
130 << "I've created one for you with default values."
131 << std::endl
132 << std::endl;
133 prm.print_parameters("parameters/heat_" + std::to_string(dim) +
134 "d.prm",
135 ParameterHandler::KeepDeclarationOrder);
136 prm.parse_input("parameters/heat_" + std::to_string(dim) + "d.prm");
137 }
138 // If something else goes wrong with parsing, then the program is
139 // terminated
140
141 // Initialization of the FunctionParser objects.
142 // Remark: when a function is time-dependent,
143 // we include "t" in the list of variables at the last spot and
144 // we set to true the fourth argument "time_dependent".
145
146 // Constants
147 std::map<std::string, double> constants;
148 constants["pi"] = numbers::PI;
149
150 // We initialize exact_solution only if it is known.
151 if (exact_solution_expression != "unknown")
152 {
153 exact_solution.initialize(
154 FunctionParser<dim>::default_variable_names() + ",t",
155 {exact_solution_expression},
156 constants,
157 true);
158 sol_is_known = true;
159 }
160 else
161 sol_is_known = false;
162
163 right_hand_side.initialize(FunctionParser<dim>::default_variable_names() +
164 ",t",
165 {rhs_expression},
166 constants,
167 true);
168 initial_value.initialize(FunctionParser<dim>::default_variable_names(),
169 {initial_expression},
170 constants);
171 boundary_value.initialize(FunctionParser<dim>::default_variable_names() +
172 ",t",
173 {boundary_expression},
174 constants,
175 true);
176 }
177
178 // Default parameters:
179
180 // Expressions for each function involved in the problem
181 std::string exact_solution_expression = "0";
182 std::string rhs_expression = "0";
183 std::string initial_expression = "0";
184 std::string boundary_expression = "0";
185
186 // FE degree
187 unsigned int fe_degree = 1;
188
189 // Global refinement steps to be performed before integration
190 unsigned int initial_refinement = 3;
191
192 // Parameters for the adaptive mesh refinement
193 double refinement_top_fraction = .3;
194 double refinement_bottom_fraction = .1;
195 double min_refinement = 1;
196 double max_refinement = 6;
197
198 // This is the gamma parameter for the penalty term in Nitsche's method
199 double gamma = 10.0;
200
201 // Parameters for the ARKode solver
202 double initial_time = 0.0;
203 double final_time = 1.0;
204 double initial_step_size = 1e-2;
205 double minimum_step_size = 1e-6;
206 unsigned int maximum_order = 5;
207 double absolute_tolerance = 1e-6;
208 double relative_tolerance = 1e-5;
209
210 // In this variable we set the number of time steps, other than
211 // initial_time, at which we want to produce a solution output
212 unsigned int out_steps = 100;
213
214 // Function objects for the functions involved in the problem.
215 // Since we will need to evaluate them at different times,
216 // we declare the time-dependent ones mutable, in order to
217 // use a const HeatParameters object in the main program.
218 mutable FunctionParser<dim> exact_solution;
219 FunctionParser<dim> initial_value;
220 mutable FunctionParser<dim> boundary_value;
221 mutable FunctionParser<dim> right_hand_side;
222
223 // Additional flag to check if the exact solution is known.
224 // If it is, we will compute the L2 error at each output time.
225 bool sol_is_known;
226
227 // mutable ParsedConvergenceTable convergence_table;
228
229 ParameterHandler prm;
230 };
