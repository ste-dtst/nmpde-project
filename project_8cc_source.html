335 template <int dim>
336 void
337 HeatEquation<dim>::assemble_ode_matrices()
338 {
339 // We assemble the mass matrix and the matrix for the implicit part
340 // of the ODE in the usual way.
341
342 // Declare the quadrature formulas
343 FEValues<dim> fe_values(fe,
344 QGauss<dim>(fe.degree + 1),
345 update_values | update_gradients |
346 update_quadrature_points | update_JxW_values);
347
348 FEFaceValues<dim> fe_face_values(fe,
349 QGauss<dim - 1>(fe.degree + 1),
350 update_values | update_gradients |
351 update_quadrature_points |
352 update_normal_vectors |
353 update_JxW_values);
354
355 // Set the usual cell-related objects
356 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
357
358 FullMatrix<double> cell_mass_matrix(dofs_per_cell, dofs_per_cell);
359 FullMatrix<double> cell_jacobian_matrix(dofs_per_cell, dofs_per_cell);
360
361 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
362
363 // Loop over cells
364 for (const auto &cell : dof_handler.active_cell_iterators())
365 {
366 // Initialize the fe_values to current cell
367 fe_values.reinit(cell);
368
369 // Initialize the cell matrices to zero
370 cell_mass_matrix = 0;
371 cell_jacobian_matrix = 0;
372
373 // The usual loop over quadrature points
374 for (const unsigned int q_index : fe_values.quadrature_point_indices())
375 for (const unsigned int i : fe_values.dof_indices())
376 for (const unsigned int j : fe_values.dof_indices())
377 {
378 cell_mass_matrix(i, j) +=
379 (fe_values.shape_value(i, q_index) * // phi_i(x_q)
380 fe_values.shape_value(j, q_index) * // phi_j(x_q)
381 fe_values.JxW(q_index)); // dx
382
383 cell_jacobian_matrix(i, j) -=
384 (fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
385 fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
386 fe_values.JxW(q_index)); // dx
387 }
388
389 // Loop over faces at the boundary of the current cell (if there are
390 // any)
391 for (auto &face : cell->face_iterators())
392 if (face->at_boundary())
393 {
394 // Initialize the fe_face_values to current face
395 fe_face_values.reinit(cell, face);
396
397 // Loop over face quadrature points. The formulas are a bit
398 // condensed, in order to do less arithmetic operations.
399 for (const unsigned int q_index :
400 fe_face_values.quadrature_point_indices())
401 for (const unsigned int i : fe_face_values.dof_indices())
402 for (const unsigned int j : fe_face_values.dof_indices())
403 cell_jacobian_matrix(i, j) +=
404 (((fe_face_values.shape_value(i, q_index) * // phi_i(x_q)
405 fe_face_values.shape_grad(
406 j, q_index) + // grad phi_j(x_q)
407 fe_face_values.shape_grad(i,
408 q_index) * // grad phi_i(x_q)
409 fe_face_values.shape_value(j,
410 q_index)) * // phi_j(x_q)
411 fe_face_values.normal_vector(q_index) - // n
412 par.gamma / (face->diameter()) * // gamma/h
413 fe_face_values.shape_value(i, q_index) * // phi_i(x_q)
414 fe_face_values.shape_value(j,
415 q_index)) * // phi_j(x_q)
416 fe_face_values.JxW(q_index)); // dx
417 }
418
419 // Initialize the local dof indices to current cell
420 cell->get_dof_indices(local_dof_indices);
421
422 // Distribute local to global
423 for (const unsigned int i : fe_values.dof_indices())
424 for (const unsigned int j : fe_values.dof_indices())
425 {
426 mass_matrix.add(local_dof_indices[i],
427 local_dof_indices[j],
428 cell_mass_matrix(i, j));
429 jacobian_matrix.add(local_dof_indices[i],
430 local_dof_indices[j],
431 cell_jacobian_matrix(i, j));
432 }
433 }
434 }

438 template <int dim>
439 void
440 HeatEquation<dim>::assemble_ode_explicit_part(const double t)
441 {
442 // We assemble also the explicit part of the ODE in the usual way.
443
444 // First, we reinit the global vector to zero
445 explicit_part.reinit(dof_handler.n_dofs());
446
447 // Declare the quadrature formulas
448 FEValues<dim> fe_values(fe,
449 QGauss<dim>(fe.degree + 1),
450 update_values | update_gradients |
451 update_quadrature_points | update_JxW_values);
452
453 FEFaceValues<dim> fe_face_values(fe,
454 QGauss<dim - 1>(fe.degree + 1),
455 update_values | update_gradients |
456 update_quadrature_points |
457 update_normal_vectors |
458 update_JxW_values);
459
460 // Set the cell-related objects
461 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
462
463 Vector<double> cell_explicit_part(dofs_per_cell);
464
465 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
466
467 // Set the right-hand side and boundary value's time to t
468 par.right_hand_side.set_time(t);
469 par.boundary_value.set_time(t);
470
471 // Loop over cells
472 for (const auto &cell : dof_handler.active_cell_iterators())
473 {
474 // Initialize the fe_values to current cell
475 fe_values.reinit(cell);
476
477 // Initialize the cell vector to zero
478 cell_explicit_part = 0;
479
480 // The usual loop over quadrature points
481 for (const unsigned int q_index : fe_values.quadrature_point_indices())
482 {
483 const auto &x_q = fe_values.quadrature_point(q_index);
484 for (const unsigned int i : fe_values.dof_indices())
485 cell_explicit_part(i) +=
486 (fe_values.shape_value(i, q_index) * // phi_i(x_q)
487 par.right_hand_side.value(x_q) * // f(x_q)
488 fe_values.JxW(q_index)); // dx
489 }
490
491 // Loop over faces at the boundary of the current cell (if there are
492 // any)
493 for (auto &face : cell->face_iterators())
494 if (face->at_boundary())
495 {
496 // Initialize the fe_face_values to current face
497 fe_face_values.reinit(cell, face);
498
499 // Loop over face quadrature points. The formula is a bit
500 // condensed, in order to do less arithmetic operations.
501 for (const unsigned int q_index :
502 fe_face_values.quadrature_point_indices())
503 {
504 const auto &x_q = fe_face_values.quadrature_point(q_index);
505 for (const unsigned int i : fe_face_values.dof_indices())
506 cell_explicit_part(i) +=
507 ((par.gamma / (face->diameter()) * // gamma/h
508 fe_face_values.shape_value(i, q_index) - // phi_i(x_q)
509 fe_face_values.shape_grad(i,
510 q_index) * // grad phi_i(x_q)
511 fe_face_values.normal_vector(q_index)) * // n
512 par.boundary_value.value(x_q) * // g(x_q)
513 fe_face_values.JxW(q_index)); // dx
514 }
515 }
516
517 // Initialize the local dof indices to current cell
518 cell->get_dof_indices(local_dof_indices);
519
520 // Distribute local to global
521 for (const unsigned int i : fe_values.dof_indices())
522 explicit_part(local_dof_indices[i]) += cell_explicit_part(i);
523 }
524 }

528 template <int dim>
529 void
530 HeatEquation<dim>::solve_ode()
531 {
532 // We set some additional data for the solver. In particular,
533 // we want to exploit the fact that the mass matrix and the
534 // Jacobian matrix are independent of time.
535 const double out_prd = (par.final_time - par.initial_time) / par.nsteps;
536 SUNDIALS::ARKode<Vector<double>>::AdditionalData data(
537 par.initial_time, // Initial time
538 par.final_time, // Final time
539 par.initial_step_size, // Initial step size
540 out_prd, // Output period
541 par.minimum_step_size, // Minimum step size
542 par.maximum_order, // Maximum order
543 10, // Maximum nonlinear iterations (irrelevant)
544 true, // implicit_function_is_linear
545 true, // implicit_function_is_time_independent
546 true, // mass_is_time_independent
547 3, // anderson_acceleration_subspace (irrelevant)
548 par.absolute_tolerance, // Absolute tolerance
549 par.relative_tolerance); // Relative tolerance
550
551 // Here we declare the ARKode object
552 SUNDIALS::ARKode<Vector<double>> ode(data);
553
554 // We have to tell the solver how to multiply the mass matrix
555 // by a vector v
556 ode.mass_times_vector =
557 [&](const double t, const Vector<double> &v, Vector<double> &Mv) {
558 (void)t;
559 mass_matrix.vmult(Mv, v);
560 };
561
562 // We set the explicit function
563 ode.explicit_function =
564 [&](const double t, const Vector<double> &y, Vector<double> &explicit_f) {
565 (void)y;
566 assemble_ode_explicit_part(t);
567 explicit_f = explicit_part;
568 };
569
570 // We set the implicit function
571 ode.implicit_function =
572 [&](const double t, const Vector<double> &y, Vector<double> &implicit_f) {
573 (void)t;
574 jacobian_matrix.vmult(implicit_f, y);
575 };
576
577 // Since the implicit function is linear, we can supply
578 // the Jacobian matrix directly to solve the implicit part
579 // of the ODE with a single Newton iteration.
580 ode.jacobian_times_vector = [&](const Vector<double> &v,
581 Vector<double> &Jv,
582 const double t,
583 const Vector<double> &y,
584 const Vector<double> &fy) {
585 (void)t;
586 (void)y;
587 (void)fy;
588 jacobian_matrix.vmult(Jv, v);
589 };
590
591 // When mass_matrix is not the identity matrix, we must
592 // provide a linear solver for the system Mx = b
593 // through ode.solve_mass.
594 // The default linear solver for SUNDIALS is a preconditioned
595 // version of GMRES. However, the mass matrix is symmetric and
596 // positive definite, therefore we can use a preconditioned
597 // conjugate gradient method. We have to specify this to the solver.
598 ode.solve_mass = [&](SUNDIALS::SundialsOperator<Vector<double>> &op,
599 SUNDIALS::SundialsPreconditioner<Vector<double>> &prec,
600 Vector<double> &x,
601 const Vector<double> &b,
602 double tol) {
603 // We forget about op and prec, since we want to provide
604 // our custom preconditioner and solver
605 (void)op;
606 (void)prec;
607
608 // We provide the solver and the preconditioner in the
609 // same way as we have done in previous tutorial programs
610 SolverControl solver_control(1000, tol);
611 SolverCG<Vector<double>> solver(solver_control);
612
613 PreconditionSSOR<SparseMatrix<double>> preconditioner;
614 preconditioner.initialize(mass_matrix, 1.2);
615
616 solver.solve(mass_matrix, x, b, preconditioner);
617 };
618
619 // Also the matrix of the linearized system is
620 // positive definite for any legal choice of gamma.
621 // Therefore, we should use CG also for the implicit part of the ODE.
622 // However, I've encountered some problems in supplying the
623 // ode.solve_linearized_system function, see temp_dummy.cc
624 // in the directory other_files.
625
626 // We supply a function for the output at fixed time intervals
627 ode.output_step = [&](const double t,
628 const Vector<double> &sol,
629 const unsigned int step_number) {
630 std::cout << "Time step " << step_number << " at t = " << t << "."
631 << std::endl;
632 std::cout << "L_inf norm of solution: " << solution.linfty_norm()
633 << std::endl;
634
635 output_results(sol, step_number);
636 };
637
638 // Finally, we are ready to solve the ODE
639 ode.solve_ode(solution);
640 }

666 DataOutBase::write_pvd_record(pvd_output, times_and_names);

671 template <int dim>
672 void
673 HeatEquation<dim>::run()
674 {
675 // Mesh generation, DoF distribution and sparsity pattern allocation
676 GridGenerator::hyper_L(triangulation);
677 triangulation.refine_global(par.initial_refinement);
678
679 setup_ode();
680
681 // Assembly of the mass matrix and the implicit part of the ODE
682 assemble_ode_matrices();
683
684 // We set the initial condition
685 VectorTools::interpolate(dof_handler, par.initial_value, solution);
686
687 // ODE solution
688 solve_ode();
689
690 // If the exact solution is known, we compute the error
691 // at par.final_time
692 if (par.sol_is_known)
693 {
694 par.exact_solution.set_time(par.final_time);
695 Vector<double> error(solution.size());
696 VectorTools::integrate_difference(dof_handler,
697 solution,
698 par.exact_solution,
699 error,
700 QGauss<dim>(fe.degree + 1),
